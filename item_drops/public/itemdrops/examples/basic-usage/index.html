using ItemDrops.Core;
using ItemDrops.Core.Drops;
using ItemDrops.Core.Types;

// Create a basic drop table
var dropTable = new DropTable(1, 3, 0.8f); // 1-3 drops, 80% chance

// Add some items
dropTable.AddDrop(new ItemDrop("sword", 2.0f)); // Weight 2.0
dropTable.AddDrop(new ItemDrop("shield", 1.5f)); // Weight 1.5
dropTable.AddDrop(new ItemDrop("potion", 3.0f)); // Weight 3.0

// Generate drops
var calculator = new DropCalculator();
var context = new LootContext { Level = 10 };
var drops = calculator.GenerateDrops(dropTable, context);

foreach (var drop in drops)
{
    Console.WriteLine($"Dropped: {drop.ItemId} x{drop.Quantity} ({drop.Rarity})");
}

using ItemDrops.Core.Conditions;

// Create a drop table with level requirements
var dropTable = new DropTable();

// Basic item - always available
dropTable.AddDrop(new ItemDrop("health_potion", 2.0f));

// Rare item - only for level 10+
var rareSword = new ItemDrop("rare_sword", 1.0f, ItemRarity.Rare)
    .WithCondition(new MinimumLevelCondition(10));

dropTable.AddDrop(rareSword);

// Boss item - only for dragons
var dragonScale = new ItemDrop("dragon_scale", 0.5f, ItemRarity.Legendary)
    .WithCondition(new EntityTypeCondition("dragon"));

dropTable.AddDrop(dragonScale);

using Godot;
using ItemDrops.Godot.Resources;
using ItemDrops.Godot.Systems;

public partial class Enemy : CharacterBody2D
{
    [Export] public DropTable DropTable { get; set; }
    
    private EnemyDrops _enemyDrops;
    
    public override void _Ready()
    {
        // Set up drop component
        _enemyDrops = new EnemyDrops();
        _enemyDrops.DropTable = DropTable;
        _enemyDrops.AutoGenerateOnDeath = true;
        _enemyDrops.DropDelay = 0.5f;
        _enemyDrops.RandomOffset = true;
        
        AddChild(_enemyDrops);
        
        // Connect to death signal
        Connect("defeated", Callable.From(OnDefeated));
    }
    
    private void OnDefeated()
    {
        // Custom death logic
        GD.Print($"{Name} defeated! Dropping items...");
        
        // Generate drops manually if needed
        // _enemyDrops.GenerateDrops();
    }
}

public partial class DropCollector : Area2D
{
    public override void _Ready()
    {
        // Monitor for item drops
        BodyEntered += OnBodyEntered;
        
        // Listen to global drop events
        ItemDropsBus.Instance.DropGenerated += OnDropGenerated;
    }
    
    private void OnBodyEntered(Node body)
    {
        if (body is ItemDropNode dropNode)
        {
            // Collect the drop
            dropNode.Collect(this);
        }
    }
    
    private void OnDropGenerated(string itemId, Vector2 position)
    {
        // Custom drop handling
        GD.Print($"Item {itemId} dropped at {position}");
        
        // Add visual effect
        var effect = GD.Load<PackedScene>("res://effects/drop_spawn.tscn").Instantiate() as Node2D;
        effect.GlobalPosition = position;
        GetTree().CurrentScene.AddChild(effect);
    }
}

// Create drop table programmatically
public static DropTable CreateEnemyDropTable()
{
    var dropTable = new DropTable();
    dropTable.MinDrops = 1;
    dropTable.MaxDrops = 3;
    dropTable.DropChance = 0.8f;
    
    // Add common drops
    var goldDrop = new DropData();
    goldDrop.ItemId = "gold";
    goldDrop.Weight = 3.0f;
    goldDrop.Quantity = new Vector2I(10, 50); // 10-50 gold
    dropTable.Drops.Add(goldDrop);
    
    // Add weapon drops with level condition
    var weaponDrop = new DropData();
    weaponDrop.ItemId = "sword";
    weaponDrop.Weight = 1.0f;
    weaponDrop.Rarity = ItemRarity.Uncommon;
    
    var levelCondition = new DropCondition();
    levelCondition.ConditionType = "MinimumLevel";
    levelCondition.IntValue = 5;
    weaponDrop.Conditions.Add(levelCondition);
    
    dropTable.Drops.Add(weaponDrop);
    
    return dropTable;
}

using UnityEngine;
using ItemDrops.Unity.Components;
using ItemDrops.Unity.Assets;

public class Enemy : MonoBehaviour
{
    [SerializeField] private DropTableAsset dropTable;
    [SerializeField] private EnemyDrops enemyDrops;
    
    private void Start()
    {
        // Configure drop behavior
        enemyDrops.DropTable = dropTable;
        enemyDrops.AutoGenerateOnDeath = true;
        enemyDrops.DropDelay = 0.5f;
        enemyDrops.RandomOffset = true;
        
        // Listen to drop events
        enemyDrops.OnDropsGenerated.AddListener(HandleDropsGenerated);
    }
    
    private void Die()
    {
        // Trigger drop generation
        enemyDrops.GenerateDrops();
    }
    
    private void HandleDropsGenerated(List<GameObject> drops)
    {
        Debug.Log($"Generated {drops.Count} drops");
        
        // Add custom effects
        foreach (var drop in drops)
        {
            var dropComponent = drop.GetComponent<ItemDropComponent>();
            if (dropComponent != null)
            {
                // Add custom behavior
                dropComponent.OnCollected.AddListener(HandleItemCollected);
            }
        }
    }
    
    private void HandleItemCollected(string itemId, GameObject collector)
    {
        Debug.Log($"{collector.name} collected {itemId}");
    }
}

using UnityEngine;
using ItemDrops.Unity.Managers;

public class CustomDropManager : MonoBehaviour
{
    [SerializeField] private DropTableAsset commonTable;
    [SerializeField] private DropTableAsset rareTable;
    
    private void Start()
    {
        // Register custom drop tables
        ItemDropsManager.Instance.RegisterDropTable("common", commonTable);
        ItemDropsManager.Instance.RegisterDropTable("rare", rareTable);
        
        // Listen to global events
        ItemDropsManager.OnDropGenerated += OnDropGenerated;
        ItemDropsManager.OnDropCollected += OnDropCollected;
    }
    
    public void GenerateLoot(Vector3 position, int playerLevel)
    {
        // Choose table based on player level
        var tableId = playerLevel > 10 ? "rare" : "common";
        
        // Create custom context
        var context = new LootContext
        {
            Level = playerLevel,
            LuckModifier = 1.0f,
            EntityType = "treasure_chest"
        };
        
        // Generate drops
        var drops = ItemDropsManager.Instance.GenerateDrops(tableId, position, context);
        
        // Add custom effects
        PlayTreasureChestEffects(position);
    }
    
    private void OnDropGenerated(string itemId, Vector3 position)
    {
        // Custom drop notification
        ShowDropNotification(itemId, position);
    }
    
    private void OnDropCollected(string itemId, GameObject collector)
    {
        // Update player inventory
        var inventory = collector.GetComponent<PlayerInventory>();
        inventory?.AddItem(itemId, 1);
    }
}

// Create drop table asset via script
[MenuItem("ItemDrops/Create Example Drop Table")]
public static void CreateExampleDropTable()
{
    var asset = ScriptableObject.CreateInstance<DropTableAsset>();
    
    // Configure basic settings
    asset.SetDropRange(1, 3);
    asset.SetDropChance(0.8f);
    
    // Add drops
    var drops = new List<DropData>();
    
    // Common items
    drops.Add(new DropData
    {
        itemId = "health_potion",
        weight = 3.0f,
        rarity = ItemRarity.Common,
        minQuantity = 1,
        maxQuantity = 2
    });
    
    // Rare items with condition
    var rareDrop = new DropData
    {
        itemId = "magic_sword",
        weight = 1.0f,
        rarity = ItemRarity.Rare,
        minQuantity = 1,
        maxQuantity = 1
    };
    
    // Add level condition
    var levelCondition = ScriptableObject.CreateInstance<DropConditionAsset>();
    levelCondition.conditionType = DropConditionType.MinimumLevel;
    levelCondition.intValue = 10;
    
    rareDrop.conditions.Add(levelCondition);
    drops.Add(rareDrop);
    
    asset.SetDrops(drops);
    
    // Save asset
    AssetDatabase.CreateAsset(asset, "Assets/Data/ExampleDropTable.asset");
    AssetDatabase.SaveAssets();
}

using Xunit;
using ItemDrops.Core.Drops;
using ItemDrops.Core.Types;

public class DropTableTests
{
    [Fact]
    public void DropTable_ShouldGenerateCorrectNumberOfDrops()
    {
        // Arrange
        var dropTable = new DropTable(2, 4, 1.0f); // Always 2-4 drops
        dropTable.AddDrop(new ItemDrop("test_item", 1.0f));
        
        var calculator = new DropCalculator(42); // Fixed seed for reproducibility
        
        // Act
        var drops = calculator.GenerateDrops(dropTable, new LootContext());
        
        // Assert
        Assert.True(drops.Count >= 2 && drops.Count <= 4);
    }
    
    [Fact]
    public void DropTable_ShouldRespectWeights()
    {
        // Arrange
        var dropTable = new DropTable(1, 1, 1.0f);
        dropTable.AddDrop(new ItemDrop("common", 9.0f)); // 90% chance
        dropTable.AddDrop(new ItemDrop("rare", 1.0f));   // 10% chance
        
        var calculator = new DropCalculator(42);
        var commonCount = 0;
        var rareCount = 0;
        
        // Act
        for (int i = 0; i < 1000; i++)
        {
            var drops = calculator.GenerateDrops(dropTable, new LootContext());
            if (drops[0].ItemId == "common") commonCount++;
            else rareCount++;
        }
        
        // Assert
        Assert.True(commonCount > 850 && commonCount < 950); // ~90%
        Assert.True(rareCount > 50 && rareCount < 150);      // ~10%
    }
}

using UnityEngine;
using UnityEngine.TestTools;
using NUnit.Framework;
using System.Collections;
using ItemDrops.Unity.Assets;

public class DropTableTests
{
    [Test]
    public void DropTableAsset_ShouldValidateCorrectly()
    {
        // Create test drop table
        var dropTable = ScriptableObject.CreateInstance<DropTableAsset>();
        
        // Add valid drops
        var drops = new List<DropData>
        {
            new DropData { itemId = "test_item", weight = 1.0f }
        };
        
        dropTable.SetDrops(drops);
        
        // Validate
        var result = dropTable.Validate();
        
        // Assert
        Assert.IsTrue(result.IsValid);
        Assert.IsEmpty(result.Errors);
    }
    
    [UnityTest]
    public IEnumerator DropTableAsset_ShouldGenerateDrops()
    {
        // Create test drop table
        var dropTable = ScriptableObject.CreateInstance<DropTableAsset>();
        dropTable.SetDropRange(1, 1);
        dropTable.SetDropChance(1.0f);
        
        // Add test drop with prefab
        var drops = new List<DropData>
        {
            new DropData 
            { 
                itemId = "test_item", 
                weight = 1.0f,
                itemPrefab = new GameObject("TestItem")
            }
        };
        
        dropTable.SetDrops(drops);
        
        // Generate drops
        var gameObjects = dropTable.GenerateDrops(Vector3.zero);
        
        // Wait for frame
        yield return null;
        
        // Assert
        Assert.AreEqual(1, gameObjects.Count);
        Assert.IsNotNull(gameObjects[0].GetComponent<ItemDropComponent>());
        
        // Cleanup
        Object.DestroyImmediate(gameObjects[0]);
    }
}

using ItemDrops.Core.Generation;
using ItemDrops.Core.Modifiers;

public class BossLootGenerator
{
    private readonly LootGenerator _generator;
    
    public BossLootGenerator()
    {
        _generator = new LootGenerator();
        
        // Add boss-specific modifiers
        _generator.AddModifier(new LuckModifier(2.0f));           // Double luck
        _generator.AddModifier(new LevelBonusModifier(50, 1.5f));  // Level 50+ bonus
        _generator.AddModifier(new RarityFilterModifier(ItemRarity.Uncommon)); // No common items
        _generator.AddModifier(new BonusItemModifier(0.3f));       // 30% chance for extra item
    }
    
    public List<DropResult> GenerateBossLoot(IDropTable baseTable, LootContext context)
    {
        var settings = new LootGenerationSettings
        {
            DropTable = baseTable,
            Context = context,
            GuaranteedDrop = true,
            MinDropsOverride = 3,
            MaxDropsOverride = 6
        };
        
        return _generator.GenerateLoot(settings);
    }
}

using ItemDrops.Core.Drops;

public class DropAnalyzer
{
    public void AnalyzeDropTable(IDropTable dropTable, int simulations = 10000)
    {
        var calculator = new DropCalculator();
        var stats = calculator.CalculateStatistics(dropTable, new LootContext(), simulations);
        
        Console.WriteLine($"Drop Table Analysis ({simulations} simulations):");
        Console.WriteLine($"Success Rate: {stats.DropSuccessRate:P2}");
        Console.WriteLine($"Average Drops: {stats.AverageDropsPerGeneration:F2}");
        Console.WriteLine($"Total Drops Generated: {stats.TotalDropsGenerated}");
        
        Console.WriteLine("\nItem Drop Frequencies:");
        foreach (var kvp in stats.ItemDropCounts.OrderByDescending(x => x.Value))
        {
            var percentage = (double)kvp.Value / stats.TotalDropsGenerated * 100;
            Console.WriteLine($"  {kvp.Key}: {kvp.Value} ({percentage:F1}%)");
        }
        
        Console.WriteLine("\nRarity Distribution:");
        foreach (var kvp in stats.RarityDropCounts.OrderByDescending(x => x.Value))
        {
            var percentage = (double)kvp.Value / stats.TotalDropsGenerated * 100;
            Console.WriteLine($"  {kvp.Key}: {kvp.Value} ({percentage:F1}%)");
        }
    }
}
