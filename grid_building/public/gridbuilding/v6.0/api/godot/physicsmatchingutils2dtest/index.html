    {
        // Given
        var container = new Node2D();
        var partialMatch = new CollisionObject2D();
        var fullMatch = new CollisionObject2D();
        var shapeCast = new ShapeCast2D();
        
        shapeCast.CollisionMask = 3; // Layers 1 and 2

        container.AddChild(fullMatch);
