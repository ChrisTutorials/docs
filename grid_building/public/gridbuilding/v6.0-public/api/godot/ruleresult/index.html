Result of a placement rule validation. Encapsulates the outcome of a rule validation including whether the rule passed, any issues encountered, and reference to the rule that was evaluated. Ported from GDScript equivalent in the placement validation system.

/// <summary>
    /// Gets a summary of the rule result.
    /// </summary>
    /// <returns>Formatted summary string</returns>
    public string GetSummary()
    {
        var summary = $"Rule {Rule.ResourcePath}: {(IsValid ? "PASS" : "FAIL")}";
        
        if (Issues.Count > 0)
        {
            summary += $"\nIssues: {string.Join(", ", Issues)}";
        }
        
        if (Notes.Count > 0)
        {
            summary += $"\nNotes: {string.Join(", ", Notes)}";
        }

        return summary;
    }
